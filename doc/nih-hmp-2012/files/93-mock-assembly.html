def do_velvet_assemble(filename, k, cwd=None):
    import os.path
    contigs_path = '%s.ass.%d/contigs.fa' % (filename, k)
    if cwd:
        contigs_path = os.path.join(cwd, contigs_path)
    
    if not os.path.exists(contigs_path):
        import subprocess
        cmd = 'bash /usr/local/src/khmer/sandbox/velvet-assemble.sh %s %d' % (filename, k)
        p = subprocess.Popen(cmd, shell=True, cwd=cwd, stdout=subprocess.PIPE, stderr=subprocess.PIPE)
        (err, out) = p.communicate()
        status = p.returncode
    
        if status != 0:
            return "%s, %s" % (out, err)
    
    logfile = '%s.ass.%d/Log' % (filename, k)
    if cwd:
        logfile = os.path.join(cwd, logfile)
        
    try:
        logdata = open(logfile).read()
        logdata = logdata.splitlines()[-1]
    except IOError:
        logdata = "error :("
    return logdata
